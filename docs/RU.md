from PIL import Image, ImageFont, ImageDraw
from pilmoji import Pilmoji
from pilmoji.source import EmojiCDNSource
import numpy as np
import pickle


def create_data(emojis,
                size=50,
                save_as=None, 
                disp=True):
    """
    –°–æ–∑–¥–∞—Ç—å —Å–ª–æ–≤–∞—Ä—å –≤—Å–µ—Ö –Ω–µ–æ–±—Ö–æ–¥–∏–º—ã—Ö –¥–∞–Ω–Ω—ã—Ö –¥–ª—è
    —Ä–∏—Å–æ–≤–∞–Ω–∏—è N —Ä–∞–∑–Ω—ã—Ö —Å–º–∞–π–ª–∏–∫–æ–≤ (–∫–æ—Ç–æ—Ä—ã–µ —É–∫–∞–∑–∞–Ω—ã
    –≤ –ø–∞—Ä–∞–º–µ—Ç—Ä–µ emojis). –°—Ç—Ä—É–∫—Ç—É—Ä–∞ —Å–ª–æ–≤–∞—Ä—è:
    {
        'emojis': list of N emojis of str type,
        'styles': dict of 4 style names {number: stylename},
        'size': int (side of square image on which emojis are drawn),
        'matrices': np-array with shape (4, N, size, size, 4) of uint8,
    }

    PARAMETERS
    ----------
    emojis : list[str]
        —Å–ø–∏—Å–æ–∫ —Å–º–∞–π–ª–∏–∫–æ–≤ –¥–ª—è —Ñ–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–∏—è –ø–∞–∫–∞

    size : int
        —Ä–∞–∑–º–µ—Ä —Å—Ç–æ—Ä–æ–Ω—ã –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è, –Ω–∞
        –∫–æ—Ç–æ—Ä–æ–º –±—É–¥–µ—Ç –æ—Ç—Ä–∏—Å–æ–≤—ã–≤–∞—Ç—å—Å—è –∫–∞–∂–¥—ã–π —Å–º–∞–π–ª–∏–∫
        (–≤ –ø–∏–∫—Å–µ–ª—è—Ö)

    save_as : str –∏–ª–∏ None
        –ø—É—Ç—å –¥–ª—è —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è –ø–∞–∫–∞. –ï—Å–ª–∏ —É–∫–∞–∑–∞—Ç—å None,
        –ø–∞–∫ –Ω–µ –±—É–¥–µ—Ç –Ω–∏–∫—É–¥–∞ —Å–æ—Ö—Ä–∞–Ω–µ–Ω, –∞ —Ç–æ–ª—å–∫–æ
        –≤–µ—Ä–Ω–µ—Ç—Å—è –≤ –∫–∞—á–µ—Å—Ç–≤–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ —Ñ—É–Ω–∫—Ü–∏–∏
    
    disp : bool
        –∏–Ω–¥–∏–∫–∞—Ü–∏—è –ø—Ä–æ–≥—Ä–µ—Å—Å–∞ —Å–æ–∑–¥–∞–Ω–∏—è –ø–∞–∫–∞
    """
    
    def _str2style(name):
        """–ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –Ω–∞–∑–≤–∞–Ω–∏–µ —Å—Ç–∏–ª—è –≤ –∫–ª–∞—Å—Å"""

            STYLE = name
        return StyleClass
        
    # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–º–∞–π–ª–∏–∫–æ–≤ –≤ –ø–∞–∫–µ

    # –†–∞–∑–º–µ—Ä –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è —Å–º–∞–π–ª–∏–∫–∞

    data_dict = {
        'emojis': emojis,
        'styles': {0: 'twitter', 
                   1: 'apple', 
                   2: 'google', 
                   3: 'facebook'},
        'size': size,
        'matrices': np.zeros((4, n_emojis, *esize, 4), 
                             dtype='uint8')
    }

    # –ö–ª–∞—Å—Å—ã –¥–ª—è —Å—Ç–∏–ª–µ–π —Å–º–∞–π–ª–∏–∫–æ–≤

    for i, emoji in enumerate(emojis):

                font = ImageFont.truetype('arial.ttf', esize[0])
                with Pilmoji(image, source=style) as pilmoji:
                    pilmoji.text((0, 0), emoji, font=font)
                    
            # RGBA-–º–∞—Ç—Ä–∏—Ü–∞
            image_np = np.array(image)
            data_dict['matrices'][j, i] = image_np
            
            # –í—ã–≤–æ–¥ –ø—Ä–æ–≥—Ä–µ—Å—Å–∞
            if disp:

                current = i * 4 + j + 1
                progress = f'{round(current / total * 100, 2)}%'
                print(f'\r{progress:<6} ({i+1}/{len(emojis)})', end='')
    if disp:
        print()

    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –ø–∞–∫–∞ –≤ –≤–∏–¥–µ –±–∏–Ω–∞—Ä–Ω–æ–≥–æ —Ñ–∞–π–ª–∞
    if save_as is not None:
        with open(f'{save_as}.pkl', 'wb') as file:
            pickle.dump(data_dict, file)

    return data_dict
